HTTP::Message::Body

Represents HTTP message body.

Constants

Default value for chunk_size

Attributes

size[R]

Size of body. nil when size is unknown (e.g. chunked response).

chunk_size[RW]

maxbytes of IO#read for streaming request. See DEFAULT_CHUNK_SIZE.

586:       def build_query_multipart_str(query, boundary)
587:         parts = Parts.new
588:         query.each do |attr, value|
589:           value ||= ''
590:           headers = ["--#{boundary}"]
591:           if Message.file?(value)
592:             remember_pos(value)
593:             param_str = params_from_file(value).collect { |k, v|
594:               "#{k}=\"#{v}\""
595:             }.join("; ")
596:             if value.respond_to?(:mime_type)
597:               content_type = value.mime_type
598:             elsif value.respond_to?(:content_type)
599:               content_type = value.content_type
600:             else
601:               path = value.respond_to?(:path) ? value.path : nil
602:               content_type = Message.mime_type(path)
603:             end
604:             headers << %{Content-Disposition: form-data; name="#{attr}"; #{param_str}}
605:             headers << %{Content-Type: #{content_type}}
606:           elsif attr.is_a?(Hash)
607:             h = attr
608:             value = h[:content]
609:             h.each do |h_key, h_val|
610:               headers << %{#{h_key}: #{h_val}} if h_key != :content
611:             end
612:             remember_pos(value) if Message.file?(value)
613:           else
614:             headers << %{Content-Disposition: form-data; name="#{attr}"}
615:           end
616:           parts.add(headers.join(CRLF) + CRLF + CRLF)
617:           parts.add(value)
618:           parts.add(CRLF)
619:         end
620:         parts.add("--#{boundary}--" + CRLF + CRLF) # empty epilogue
621:         parts
622:       end
